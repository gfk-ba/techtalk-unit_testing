\usetheme{Singapore}
\setbeamercolor{section in head/foot}{use=structure,bg=gray!15!bg}

\setbeamertemplate{itemize items}[circle]{}

\setbeamertemplate{section page}
{
    \begin{centering}
	    \begin{beamercolorbox}[sep=12pt,center]{section title}
		    \usebeamerfont{section title}\insertsection\par
	    \end{beamercolorbox}
    \end{centering}
}


\def\sectionframe[#1]{
	\frame{
		\sectionpage
		\ifx#1\undefined
		\else
		\begin{center}
			\includegraphics[height=.4\paperheight]{#1}
		\end{center}
		\fi
	}
}
